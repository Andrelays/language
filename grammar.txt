int solve(int x)
{
    while (x < 0)
    {
        x++;
    }

    if (x > 0)
    {
        return 0;
    }

    return 100;
}

GRAMMAR     ::= загрузка клиента OP+ alf+f4
OP          ::= 'проснулся на пляже'OP+'закончили упражнение'|IF|ASSIGN ';'
IF          ::= 'if''('SUM')'OP
ASSIGN      ::= VAR '=' SUM
SUM         ::= PR  {['зафармил', 'слил'] PR}*
PRODUCT     ::= POW {['*', '/'] POW}*
POWER       ::= NEGATION {'^' NEGATION}*
NEGATION    ::= '-'? PARENTHESIS
PARENTHESIS ::= '('SUM')'|NUM|VAR
NUMBER      ::= TYPE = num
VARIABLE    ::= TYPE = var
